/** Base resource specification. */
Specification = Object {
  // List of persons that will use the resource in videoconference.
  persons: List<1:n> of Person,

  // Option specifying who should initiate the call.
  callInitiation: Enum{Default, Terminal, Multipoint},
}

/** Fully-qualified explicit specification. */
ResourceSpecification = Object extends Specification {
  // Specific resource.
  resource: Resource,
} 

/** Partially-qualified explicit specification. */
LookupSpecification = Object extends Specification {
  // Technology of a resource.
  technology: * or Technology,
  // Type of a resource.
  type: Enum {Terminal, Multipoint}
}

/** Terminal specification. */
TerminalSpecification = Object extends Specification {
  // Technology of a terminal.
  technology: Technology,
  // Alias of a terminal.
  alias: Alias or IdentifierResource,
  // Number of terminals with the same definition (default 1).
  count: Integer
}

/** Virtual room specification. */
VirtualRoomSpecification = Object extends Specification {
  // Device for the virtual room.
  device: DeviceResource,
  // Number of participants in the virtual room.
  size: Integer
}    

/** 
 * Single reservation that was requested by an user.
 */
Reservation = Object {
  // Simple or periodic date.
  dateTime: DateTime,

  // Requested resources for the reservation.
  requestedResources: List<0:n> of Specification,

  // List of persons that should choose resource for the reservation.
  requestedPersons: List<0:n> of Person,

  // Option specifying who should initiate the call.
  callInitiation: Enum{Default, Terminal, Multipoint},

  // Option specifying whether inter domain lookup can be performed.
  interDomain: Boolean
}
